// Configure once at startup
marked.use({ gfm: true, breaks: true });

/**
 * Re-render accumulated markdown into a container element.
 * Call this on every incoming chunk from the LLM stream.
 *
 * @param {string}      partial   - Full accumulated markdown so far
 * @param {HTMLElement} container - The DOM element to render into
 */
function renderMarkdown(partial, container) {
  container.innerHTML = marked.parse(partial);

  // Syntax-highlight every code block
  container.querySelectorAll('pre code').forEach(block => {
    hljs.highlightElement(block);
  });
}

// Inject chunks as they arrive:
let accumulated = '';

function onChunk(chunk) {
  accumulated += chunk;
  renderMarkdown(accumulated, document.getElementById('output'));
}

// On stream end, do a final clean render (no cursor):
function onDone() {
  renderMarkdown(accumulated, document.getElementById('output'));
}
// Anthropic streaming via fetch (works with any OpenAI-compatible API too)
async function streamCompletion(prompt, outputEl) {
  let accumulated = '';

  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {
      'Content-Type':         'application/json',
      'x-api-key':            YOUR_API_KEY,
      'anthropic-version':    '2023-06-01',
    },
    body: JSON.stringify({
      model:      'claude-opus-4-6',
      max_tokens: 2048,
      stream:     true,
      messages:   [{ role: 'user', content: prompt }],
    }),
  });

  const reader  = res.body.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    const lines = decoder.decode(value).split('\n');
    for (const line of lines) {
      if (!line.startsWith('data:')) continue;
      try {
        const json = JSON.parse(line.slice(5).trim());
        const delta = json?.delta?.text ?? '';
        if (!delta) continue;

        accumulated += delta;
        renderMarkdown(accumulated, outputEl); // <-- your render fn
      } catch { /* partial JSON line, skip */ }
    }
  }

  // Final clean render
  renderMarkdown(accumulated, outputEl);
}
